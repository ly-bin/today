##### 1、题目：https://leetcode-cn.com/problems/convert-binary-number-in-a-linked-list-to-integer/。   中等 
##### 2、思路：
> 1、翻转链表，按位相加，
> 
> 2、数学法

##### 3、代码
```
public class GetDecimalValue {

    public class ListNode {
        int val;
        ListNode next;

        ListNode() {
        }

        ListNode(int val) {
            this.val = val;
        }

        ListNode(int val, ListNode next) {
            this.val = val;
            this.next = next;
        }
    }

    public int getDecimalValue1(ListNode head) {
        if (head == null) {
            return 0;
        }
        int count = 0;
        ListNode indexNode = head;
        while (indexNode != null) {
            indexNode = indexNode.next;
            count++;
        }
        int sum = 0;
        while (head != null) {
            count--;
            if (head.val == 1) {
                sum += head.val * Math.pow(2, count);
            }
            head = head.next;
        }
        return sum;
    }

    public int getDecimalValue(ListNode head) {
        if (head == null) {
            return 0;
        }
        ListNode preNode = null;
        while (head != null) {
            ListNode temp = head.next;
            head.next = preNode;
            preNode = head;

            head = temp;
        }
        int index = 0;
        int sum = 0;
        while (preNode != null) {
            if (preNode.val == 1) {
                sum += preNode.val * Math.pow(2, index);
            }
            preNode = preNode.next;
            index++;
        }
        return sum;
    }
}
```
